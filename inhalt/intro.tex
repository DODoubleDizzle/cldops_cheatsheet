\section{General Definitions}
\textbf{NIST Cloud:} On-demand self-service, Broad network access, Resource pooling (multi-tenant), Rapid elasticity, Measured service (pay-as-you-go).
\textbf{Cloud Operations:} manage/optimize cloud services \& infra.
\textbf{GitOps:} Git = single source of truth; declarative, versioned CI/CD.
\textbf{DevOps:} dev+ops; automation, CI/CD, monitoring, agile.
\textbf{CI:} every push auto-built \& tested.
\textbf{Platform Engineering:} internal self-service dev platforms (IDP); faster shipping.
\textbf{DevOps 6 pillars:} Cultural mindset, Automation-first, Continuous Processes, Monitoring/Feedback, Rapid+Reliable Releases, Security.
\subsection{DevOps Cycle}
\includegraphics[width=\linewidth]{devops.png} \\
\textbf{Arcs $\to$ stages:} \textcolor{ciGreen}{\textbf{CI:}} Code+Build+Test, \textcolor{cdBlue}{\textbf{CDeployment:}} Release+Deploy (+Test), \textcolor{deliveryPurple}{\textbf{CDelivery:}} Test+Release (manual prod), \textcolor{feedbackOrange}{\textbf{CFeedback:}} Operate+Monitor+Plan. \\
\textbf{Stages (\actv{activities} | \tool{tools}):}
\textbf{Plan} (\actv{requirements, roadmap, arch sketch, risk/rollback} | \tool{Jira, Asana, GitHub Issues, Trello}),
\textbf{Code} (\actv{write/refactor, local build/test, peer review} | \tool{GitLab, GitHub, Bitbucket}),
\textbf{Build} (\actv{compile, container images, publish artifacts, versioned packages} | \tool{GitLab CI, GH Actions, Jenkins, Maven/Gradle, Docker/Podman, Artifactory, Quay}),
\textbf{Test} (\actv{unit/integration/E2E, perf/load, sec scans, coverage, quality gates} | \tool{JUnit/pytest, Selenium/Cypress/Playwright, JMeter/k6, Snyk/Trivy, SonarQube}),
\textbf{Release} (\actv{promote artifact, immutable version, release notes, signing} | \tool{Cosign, GnuPG, GitLab/GH Packages}),
\textbf{Deploy} (\actv{pull artifact, apply manifest, strategy, post-deploy validation, rollback} | \tool{ArgoCD, Flux, Helm, Kustomize, Terraform/OpenTofu, Ansible, Flagger}),
\textbf{Operate} (\actv{troubleshoot, scale, patch, secret rotation, drift detect, backups, compliance} | \tool{PagerDuty, Vault, Velero, HPA/VPA, Falco}),
\textbf{Monitor} (\actv{SLI/SLO, dashboards, alerts, synthetic checks, correlate logs/metrics/traces} | \tool{Prometheus, Grafana, Loki, Jaeger, ELK, Alertmanager}). \\
\textbf{CDelivery vs CDeployment:} Deployment auto-pushes staging$\to$prod; Delivery manual gate. \\
\textbf{CD Deployment Strategies:}
\begin{itemize}
	\item \textbf{Rolling:} gradual update, minimal downtime
	\item \textbf{Blue-Green:} two envs, fast switch
	\item \textbf{Canary:} small user group first
	\item \textbf{Feature Flag:} deploy, toggle later
	\item \textbf{Dark Launching:} real traffic, \\ hidden from users
\end{itemize}
\subsection{Platform Engineering}
\textbf{Core Responsibilities:} enable self-service, standardize runtimes/tooling, abstract infra, enforce compliance \& security, provide observability. \\
\textbf{8 Building Blocks (top$\to$bottom):} Self-Service Portal (IDP), Cost \& Capacity, Observability, Policy \& Security, Package \& Artifact Mgmt, CI/CD Pipelines \& Deployment, Infrastructure as Code, Compute \& Orchestration. \\
\textbf{Top Container Challenges (CNCF 2025):} Cultural changes (47\%), Lack of training (36\%), Security (36\%), CI/CD (35\%), Monitoring (35\%), Complexity (34\%), Scaling, Networking, Logging, Service Mesh.
